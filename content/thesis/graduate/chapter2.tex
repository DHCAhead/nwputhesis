\section{公式排版示例}
这里给出正文中的公式引用示例。行内公式可写为 $E=mc^2$，独立编号公式如\autoref{eq:graduate-quadratic-formula} 所示，多行推导公式如\autoref{eq:graduate-gaussian-sum} 所示。

\begin{equation}
    x = \frac{-b \pm \sqrt{b^2 - 4ac}}{2a}
    \label{eq:graduate-quadratic-formula}
\end{equation}
\begin{eqvariables}
    式中：$x$ 为一元二次方程的根，$a$、$b$、$c$ 为方程系数，且 $a \neq 0$。
\end{eqvariables}

\begin{equation}
    \begin{aligned}
        \sum_{i=1}^{n} i &= 1 + 2 + \cdots + n \\
                         &= \frac{n(n+1)}{2}
    \end{aligned}
    \label{eq:graduate-gaussian-sum}
\end{equation}
